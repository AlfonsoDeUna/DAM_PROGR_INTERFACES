```xml
        <dependency>
            <groupId>org.junit.jupiter</groupId>

        </dependency>
        <dependency>
            <groupId>org.junit.platform</groupId>
            <artifactId>junit-platform-runner</artifactId>

        </dependency>    
    </dependencies>

```
